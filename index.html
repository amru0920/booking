const SHEET_NAME = 'Bookings'; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ 'ID', 'Matric', 'Name', 'Court', 'Date', 'Slot', 'Duration', 'Status', 'Timestamp' ]); sheet.getRange(1, 1, 1, 9).setFontWeight('bold'); } let action = 'get'; let body = {}; try { if (e.postData && e.postData.contents) { body = JSON.parse(e.postData.contents); action = body.action || 'get'; } else if (e.parameter && e.parameter.action) { action = e.parameter.action; } } catch(err) {} if (action === 'add') { const id = 'BK' + Date.now(); sheet.appendRow([ id, body.matric, body.name || '', body.court, body.date, body.slot, body.duration, 'Active', new Date().toLocaleString('ms-MY') ]); return output({ success: true, id: id }); } if (action === 'get') { const matric = e.parameter ? e.parameter.matric : ''; const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return output({ success: true, data: [] }); const headers = rows[0]; let data = rows.slice(1).map(r => { let obj = {}; headers.forEach((h, i) => { obj[h] = r[i]; }); return obj; }); if (matric) data = data.filter(d => d.Matric === matric); return output({ success: true, data: data }); } if (action === 'cancel') { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == body.id) { sheet.getRange(i + 1, 8).setValue('Cancelled'); break; } } return output({ success: true }); } if (action === 'check') { const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return output({ success: true, booked: [] }); const headers = rows[0]; const data = rows.slice(1).map(r => { let obj = {}; headers.forEach((h, i) => { obj[h] = r[i]; }); return obj; }); const booked = data .filter(d => d.Court === e.parameter.court && d.Date === e.parameter.date && d.Status === 'Active') .map(d => d.Slot); return output({ success: true, booked: booked }); } return output({ success: false, error: 'Unknown action' }); } function output(data) { const json = JSON.stringify(data); return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
